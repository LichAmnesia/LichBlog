bool bfs(int s,int flag){
	while(!q.empty()) q.pop();
	q.push(s);
	int cnt = -1;
	while(!q.empty()){
		int k = q.front(); q.pop();
		cnt++;
		for (int i = 0; i < v[k].size(); i++){
			int d = v[k][i];
			if (!vis[d] && num[d] == flag){
				vis[d] = 1;
				q.push(d);
			}
		}
	}
	if(flag == 0 && cnt == 0)return 1;
	return 0;
}
